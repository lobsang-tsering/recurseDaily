class Solution {
    public int longestOnes(int[] nums, int k) {
       int L = 0;
       int max = 0;
       int count = 0;
       for(int R = 0; R < nums.length; R++) {
        if(nums[R] == 0) {
            count++;
        }

        while(count > k) {
            if(nums[L] == 0) {
                count--;
            }
            L++;
        }
        max = Math.max(max, R - L + 1);
       }
       return max;
    }
}
