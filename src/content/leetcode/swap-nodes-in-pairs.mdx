```java
class Solution {
    public ListNode swapPairs(ListNode head) {
        if(head == null || head.next == null) return head;
        ListNode p = head.next;
        ListNode rest = p.next;
        p.next = head;
        head.next = swapPairs(rest);
        return p;
    }
}
```
