```java

class Solution {
    public int minSubArrayLen(int target, int[] nums) {
        int L = 0; 
        int sum = 0;
        int result = Integer.MAX_VALUE;
        for (int R = 0;  R < nums.length; R++) {
            sum += nums[R];

            while(sum >= target) {
                result = Math.min(R - L + 1, result);
                sum -= nums[L++];
            }
        }
        return result == Integer.MAX_VALUE ? 0: result; 
    }
}
```
