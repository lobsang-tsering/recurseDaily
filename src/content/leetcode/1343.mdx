Class Solution {
    public int numOfSubarrays(int[] arr, int k, int threshold) {
        int i = 0;
        int j = 0;
        int windowSum = 0;
        int counter = 0;

        while(j < arr.length) {
            windowSum += arr[j];
            if(j - i + 1 < k) {
                j++
            } else if ( j - i + 1 == k) {
                int avg = windowSum/k;

                if(avg > threshold) counter++;

                windowSum -= arr[i];
                i++;
                j++;
            }
        }
        return counter;
    }
}
