```java
class Solution {
    public boolean containsNearbyDuplicate(int[] nums, int k) {
       HashSet<Integer> window = new HashSet<Integer>();
       int L = 0;

       for(int R = 0; R < nums.length; R++) {
         if(R - L > k) {
            window.remove(nums[L]);
            L++;
         }

         if(!window.add(nums[R])) return true;
       }

       return false;
    }
}
```
